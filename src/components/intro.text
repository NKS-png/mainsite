        <svg viewBox="0 0 600 200">
            <text x="50%" y="50%" text-anchor="middle" dy=".35em" class="sketch-text" id="logo-text">
                NKScreate

    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const textElement = document.getElementById('logo-text');
            const underlineElement = document.getElementById('underline');
            const introOverlay = document.getElementById('intro-overlay');
            const mainContent = document.getElementById('main-content');
            const body = document.body;

            // 1. Setup the Text Animation (Stroke)
            // We calculate the length of the text path to animate it perfectly
            const textLength = textElement.getComputedTextLength();
            
            // Set up CSS variables for the stroke
            textElement.style.strokeDasharray = textLength;
            textElement.style.strokeDashoffset = textLength;
            
            // Force a layout reflow (flush CSS)
            textElement.getBoundingClientRect();

            // 2. Setup Underline Animation
            const underlineLength = underlineElement.getTotalLength();
            underlineElement.style.strokeDasharray = underlineLength;
            underlineElement.style.strokeDashoffset = underlineLength;
            underlineElement.style.opacity = '1';

            // 3. EXECUTE ANIMATION TIMELINE
            
            // Start writing text (Duration: 2s)
            textElement.style.transition = 'stroke-dashoffset 2s ease-in-out, fill 1s ease 2s';
            textElement.style.strokeDashoffset = '0';

            // Start writing underline (Start at 1.8s, Duration 0.5s)
            setTimeout(() => {
                underlineElement.style.transition = 'stroke-dashoffset 0.5s ease-out';
                underlineElement.style.strokeDashoffset = '0';
            }, 1800);

            // Fill the text with color (At 2.2s)
            setTimeout(() => {
                textElement.style.fill = '#222'; // Fade in the black fill
            }, 2200);

            // 4. EXIT INTRO (At 3.5s)
            setTimeout(() => {
                introOverlay.classList.add('fade-out');
                mainContent.classList.add('content-visible');
                body.style.overflow = 'auto'; // Re-enable scrolling
            }, 3500);
        });
    </script>
